貸出記録テーブル(rental)に対して、以下の内容で新規データを挿入してみましょう

項目	値
利用者コード	B200502
ISBNコード	4-0010-0000-0
貸出日	（今日の日付）

問題
①
	rental
	(
		user_id,
		isbn,
		rental_date
	)
②
	(
		'B200502',
		'4-0010-0000-0',
		CURRENT_DATE
	)
;

回答
-- 挿入命令
INSERT INTO
	-- 貸出記録テーブル
	rental
	(
		-- 利用者コードカラム
		user_id,
		-- ISBNコードカラム
		isbn,
		-- 貸出記録カラム
		rental_date
	)
-- 挿入値
VALUES
	(
		-- 利用者コードカラム	
		'B200502',
		-- ISBNコードカラム
		'4-0010-0000-0',
		-- 貸出記録カラム
		CURRENT_DATE
	)
;

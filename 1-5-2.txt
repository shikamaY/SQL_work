取り出す列は、name列のみとします。

-- 取得命令
SELECT
-- 名前カラム
	name
-- 取得元
FROM
-- 回答テーブル
	quest
-- 取得条件
WHERE
-- 名前に子を含まない
	name NOT LIKE '%子'	
;

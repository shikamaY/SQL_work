アンケート集計テーブル(quest)から年齢(age列)が30以上40未満の人の回答のみを抜き出してみましょう。
取り出す列は,name,sex,prefecture列とします。

-- 取得命令
SELECT
-- 氏名カラム
	name,
-- 性別カラム
	sex,
-- 都道府県カラム
	prefecture
-- 取得元
FROM
-- アンケート回答テーブル
	quest
-- 取得条件
WHERE
-- 年齢が20以上
	age BETWEEN 30 AND 39
;

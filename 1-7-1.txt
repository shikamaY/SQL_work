書籍情報テーブル(books)から価格が2500〜3500円の範囲の書籍を価格が安い順に取り出してみましょう。
取り出す列は,title、price列とします。

-- 取得命令
SELECT
-- 書籍名カラム
	title,
-- 値段カラム
	price
-- 取得元
FROM
-- 書籍テーブル
	books
-- 取得条件
WHERE
-- 価格が2500〜3500
	price BETWEEN 2500 AND 3500
-- 昇順でソート	
ORDER BY
-- 価格
	price ASC
;

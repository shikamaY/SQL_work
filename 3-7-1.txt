貸出記録テーブル(rental)から貸出日が2011年12月31以前で、かつ、
返却済みである情報について削除してみましょう。空白を埋めて、
正しいSQL命令を完成させてください。

問題
①
	rental
WEHRE
	②
;

回答
-- 削除命令
DELETE FROM
	-- 貸出記録テーブル
	rental
-- 削除条件
WHERE
	-- 貸出日が2011年12月31以前
	rental_date < '2011-12-31'
	-- 返却済み
	AND returned = '1'
;

kensyu-# -- 削除条件

kensyu-#        rental_date < '2011-12-31'
kensyu-#        -- 返却済み
kensyu-#        AND returned = '1'
kensyu-# ;

以下は、社員テーブル(employee)から部署、役職ごとに降順ソートした結果を出力するための
SQL命令ですが、誤りが2点あります。誤りを指摘してください。

問題
SELECT
	depart_id,
	class,
	CONCAT('l_name','f_name') AS 氏名
FROM
	employee
ORDER BY
	depart_id,
	class DESC
;

回答
-- 取得命令
SELECT
	-- 部署カラム
	depart_id,
	-- 役職カラム
	class,
	-- 名字、名前を結合して別名表示 ⇒ 問題では''で囲っており、カラム名ではなく文字列として結合されていた
	l_name || f_name AS 氏名
-- 取得元
FROM
	-- 社員テーブル
	employee
-- ソート
ORDER BY
	-- 部署カラムを降順 ⇒ 問題では降順を指定していなかったため、昇順で表示されていた
	depart_id DESC,
	-- 役職カラムを降順
	class DESC
;

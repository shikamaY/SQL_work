アンケート回答テーブル(quest)から都道府県別の評価の平均値を求めて見ましょう。
その際、平均値は小数点以下を四捨五入します。以下の空欄を埋めて、SQL命令を完成させてください。

問題
SELECT
	prefecture,
	① AS 評価平均
FROM
	quest
GROUP BY
	②
;

回答
-- 取得命令
SELECT
	-- 都道府県カラム
	prefecture,
	-- 評価の平均値を求め、小数点以下を四捨五入し別名で表示
	ROUND(AVG(answer1),0) AS 評価平均
-- 取得元
FROM
	-- アンケート回答テーブル
	quest
-- グループ化
GROUP BY
	-- 都道府県ごとに
	prefecture
;

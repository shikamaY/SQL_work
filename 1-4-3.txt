ユーザテーブル(usr)から「東京」に住んでいない人の情報だけを取り出してみましょう。
取り出す列は、l_name,f_name,email列とします。

-- 取得命令
SELECT 
-- 苗字カラム
	l_name,
-- 名前カラム
	f_name,
-- e-メールカラム
	email
-- 取得元
FROM
-- ユーザテーブル
	usr
-- 取得条件
WHERE
-- 都道府県に「東京」を含まない
	prefecture <> '東京都'
;

以下は、貸出記録テーブル（rental）上、利用者のコードがB200405,ISBNコード
が4-0010-0000-1のデータについて、貸出日を今日の日付に更新するためのSQL命令ですが、
誤りが2点あります。誤りを指摘してください。

問題
UPDATE
	rental
SET
	rental_date CURRENT_DATE
WHERE
		user_id='B200405'
	OR
		isbn='4-0010-0000-1'
;

回答
-- 更新命令
UPDATE
	-- 貸出記録テーブル
	rental
-- 更新値セット
SET
	-- 貸出日を実行日日付に変更、誤：「=」が抜けている
	rental_date = CURRENT_DATE
-- 更新条件
WHERE
		-- ユーザIDが「B200405」
		user_id='B200405'
	-- アンド条件、誤:「OR」になっている
	AND
		-- ISBNコードが「4-0010-0000-1」
		isbn='4-0010-0000-1'
;

kensyu=# -- 更新命令
kensyu=# UPDATE

kensyu-#        rental

kensyu-#        -- 貸出日を実行日日付に変更、誤：「=」が抜けている
kensyu-#        rental_date = CURRENT_DATE
kensyu-# -- 更新条件
kensyu-# WHERE
kensyu-#                -- ユーザIDが「B200405」
kensyu-#                user_id='B200405'
kensyu-#        -- アンド条件、誤:「OR」になっている
kensyu-#        AND
kensyu-#                -- ISBNコードが「4-0010-0000-1」
kensyu-#                isbn='4-0010-0000-1'
kensyu-# ;

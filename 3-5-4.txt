貸出記録テーブル（rental)上、貸出日が2012年3月31日より前で、
かつ、現在貸出中であるレコードについてreturned列を９（紛失）で更新しましょう。

-- 更新命令
UPDATE
	-- 貸出記録テーブル
	rental
-- 更新値セット
SET
	-- 貸出状態カラムを変更
	returned = 9
-- 更新条件
WHERE
	-- 貸出日が「2012-03-31」より前
	rental_date < '2012-03-31'
	-- 貸出中の場合
	AND returned = 0
;

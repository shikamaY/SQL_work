貸し出し記録テーブル（rental)から未返却で、貸出日が2012年12月1日より前の貸し出し情報を、
貸出日が新しい順に抽出してみましょう。抽出する対象列は全列とします。

-- 取得命令
SELECT
-- 全カラム
	*
-- 取得元
FROM
-- 貸し出し記録テーブル
	rental
-- 取得条件
WHERE
-- 貸出日が2012年12月1日より前
	rental_date <= '2012-12-01'
-- かつ
AND
-- 未返却
	returned = 0
-- ソート
ORDER BY
-- 貸出日を昇順
	rental_date ASC
;

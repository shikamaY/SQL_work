ユーザテーブル(usr)から東京都,千葉県,神奈川県に住んでいる人のリストを姓（カナ),名(カナ)について
昇順でリストしてみましょう。取り出す列は、l_name,f_name,prefecture列とします。

-- 取得命令
SELECT
-- 名字カラム
	l_name,
-- 名前カラム
	f_name,
-- 都道府県カラム
	prefecture
-- 取得元
FROM
-- ユーザテーブル
	usr
-- 取得条件
WHERE
-- 都道府県が東京都、千葉県、神奈川県の場合
	prefecture IN ('東京都','千葉県','神奈川県')
-- ソート条件
ORDER BY
-- 名字（カナ）を昇順
	l_name_kana ASC,
-- 名前（カナ）を昇順
	f_name_kana ASC
;
